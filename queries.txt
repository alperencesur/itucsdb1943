CREATE TABLE Post (
POSTID SERIAL PRIMARY KEY,
USERID INTEGER,
POSTDATE VARCHAR(24),
PHOTOURL VARCHAR(255) NOT NULL,
DESCRIPTION VARCHAR(255),
TITLE VARCHAR(255) NOT NULL,
SCORE VARCHAR(255),
TAG VARCHAR(20)
)

CREATE TABLE Users (
USERID SERIAL PRIMARY KEY,
NAME VARCHAR(40) NOT NULL,
SURNAME VARCHAR(40) NOT NULL,
EMAIL VARCHAR(255),
DESCRIPTION VARCHAR(255),
TITLE INTEGER,
SCORE VARCHAR(255),
TAG VARCHAR(20)
)
